[Unit]
Description=gunicorn daemon
Requires=gunicorn.socket
After=network.target

[Service]
User=ubuntu
Group=www-data
WorkingDirectory=/home/ubuntu/pyapps/landnet
ExecStart=/home/ubuntu/pyapps/venv/bin/gunicorn \
          --access-logfile - \
          --workers 3 \
          --bind unix:/run/gunicorn.sock \
          landnet.wsgi:application

[Install]
WantedBy=multi-user.target


sudo nano /etc/nginx/sites-available/landnet


sudo ln -s /etc/nginx/sites-available/landnet /etc/nginx/sites-enabled
